Tank Game

Engine: Unity

Role: Solo Developer

Development Description: I created this game alone. Art was imported or simple shapes but; mechanics, movement, animation, sound, particle effects, UI, and AI were done by me. This was a school project and the first time I had used Unreal.

Game Description: Shoot the other tank in the arena without dying.

Enemy AI code snippet 

void Update()
{								
	if(alive) 
	{
		//get sight vector to player & set navmesh agent
		Vector3 sight = player.transform.position - transform.position;
		nav_mesh_agent = GetComponent();

		// if in range start moving and firing
		if (sight.magnitude < vis_distance)
		{
			nav_mesh_agent.destination = move_pos_transform.position;
			transform.rotation = transform.rotation * Quaternion.Euler(-90, 0.01f, 0.01f);
			//fire shot
			if (fire_cd <= 0)
			{
				Quaternion brot = transform.rotation * Quaternion.AngleAxis(180, Vector3.up);
				GameObject temp = Instantiate(bullet_prefab, transform.position, brot);
				temp.tag = "enemy_bullet";
				fire_cd = set_fire_cd;
			}
			else
				fire_cd -= 2;
		}
	}
}
